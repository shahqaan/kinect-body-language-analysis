12 /* Skeletons sets up four 'observers' (listeners) so that
13  when a new user is detected in the scene, a standard pose for that
14  user is detected, the user skeleton is calibrated in the pose, and then the
15  skeleton is tracked. The start of tracking adds a skeleton entry to userSkels.
16 
17  Each call to update() updates the joint positions for each user's
18  skeleton.
19 
20  Each call to draw() draws each user's skeleton, with a rotated HEAD_FNM
21  image for their head, and status text at the body's center-of-mass.
22  */

114  * Update the skeleton of each user

116  */
117  public void update() {
118  try {
119  context.waitAnyUpdateAll();
120  } catch (StatusException e) {
121  System.out.println(e);
122  System.exit(1);
123  }
124  try {
125  int[] userIDs = userGen.getUsers(); // there may be many users in the scene
126  for (int i = 0; i < userIDs.length; ++i) {
127  int userID = userIDs[i];
128  if (skelCap.isSkeletonCalibrating(userID)) {
129  continue; // test to avoid occassional crashes with isSkeletonTracking()
130  }
131  if (skelCap.isSkeletonTracking(userID)) {
132  updateJoints(userID);
133  }
134  }
135  } catch (StatusException e) {
136  System.out.println(e);
137  }

178  protected void updateJoint(HashMap<SkeletonJoint, SkeletonJointPosition> skel,
179  int userID, SkeletonJoint joint) {
180  try {
181  // report unavailable joints (should not happen)
182  if (!skelCap.isJointAvailable(joint) || !skelCap.isJointActive(joint)) {
183  System.out.println(joint + " not available for updates");
184  return;
185  }
186 
187  SkeletonJointPosition pos = skelCap.getSkeletonJointPosition(userID, joint);
188  if (pos == null) {
189  System.out.println("No update for " + joint);
190  return;
191  }
192 
193  /*
194  // the call to getSkeletonJointOrientation() crashes Java!
195  SkeletonJointOrientation ori = skelCap.getSkeletonJointOrientation(userID, joint);
196  if (ori == null)
197  System.out.println("No orientation for " + joint);
198  else
199  System.out.println("Orientation for " + joint + ": " + ori);
200  */
201  SkeletonJointPosition jPos = null;
202  if (pos.getPosition().getZ() != 0) // has a depth position
203  {
204  jPos = new SkeletonJointPosition(
205  depthGen.convertRealWorldToProjective(pos.getPosition()),
206  pos.getConfidence());
207  } else // no info found for that user's joint
208  {
209  jPos = new SkeletonJointPosition(new Point3D(), 0);
210  }
211  skel.put(joint, jPos);
212  } catch (StatusException e) {
213  System.out.println(e);
214  }
215  }
216  protected Point3D getJointPos(HashMap<SkeletonJoint, SkeletonJointPosition> skel,
217  SkeletonJoint j) // get the (x, y, z) coordinate for the joint (or return null)
218  {
219  SkeletonJointPosition pos = skel.get(j);
220  if (pos == null) {
221  return null;
222  }
223 
224  if (pos.getConfidence() == 0) {
225  return null; // don't draw a line to a joint with a zero-confidence pos
226  }
227  return pos.getPosition();
228  }

256  class PoseDetectedObserver implements IObserver<PoseDetectionEventArgs> {
257 
258  public void update(IObservable<PoseDetectionEventArgs> observable,
259  PoseDetectionEventArgs args) {
260  int userID = args.getUser();
261  System.out.println(args.getPose() + " pose detected for user " + userID);
262  try {
263  // finished pose detection; switch to skeleton calibration
264  poseDetectionCap.StopPoseDetection(userID); // big-S ?
265  skelCap.requestSkeletonCalibration(userID, true);
266  } catch (StatusException e) {
267  e.printStackTrace();
268  }

272  class CalibrationCompleteObserver implements IObserver<CalibrationProgressEventArgs> {
273 
274  @Override
275  public void update(IObservable<CalibrationProgressEventArgs> observable,
276  CalibrationProgressEventArgs args) {
277  int userID = args.getUser();
278  System.out.println("Calibration status: " + args.getStatus() + " for user " + userID);
279  try {
280  if (args.getStatus() == CalibrationProgressStatus.OK) {
281  // calibration succeeeded; move to skeleton tracking
282  System.out.println("Starting tracking user " + userID);
283  skelCap.startTracking(userID);
284  userSkels.put(new Integer(userID),
285  new HashMap<SkeletonJoint, SkeletonJointPosition>());
286  // create new skeleton map for the user in userSkels
287  } else // calibration failed; return to pose detection
288  {
289  poseDetectionCap.StartPoseDetection(calibPoseName, userID); // big-S ?
290  }
291  } catch (StatusException e) {
292  e.printStackTrace();
293  }
