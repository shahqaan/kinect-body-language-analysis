141  * @param z
142  * @param a

146  * @param q

148  * @return
149  * angle in radians
150  */
151  private double findAngle(
152  float x, float y, float z,
153  float a, float b, float c,
154  float p, float q, float r) {
155 
156  float i, j, k;
157  i = ((y - b) * (r - z)) - ((z - c) * (q - y));
158  j = ((x - a) * (r - z)) - ((z - c) * (p - x));
159  k = ((x - a) * (q - y)) - ((y - b) * (p - x));
160 
161  i = i * i;
162  j = j * j;
163  k = k * k;
164 
165  double numerator = Math.sqrt(i + j + k);
166 
167  float v1 = (x-a)*(x-a);
168  float v2 = (y-b)*(y-b);
169  float v3 = (z-c)*(z-c);
170 
171  float u1 = (p-x)*(p-x);
172  float u2 = (q-y)*(q-y);
173  float u3 = (r-z)*(r-z);
174 
175  double denom1 = Math.sqrt(v1 + v2 + v3);
176  double denom2 = Math.sqrt(u1 + u2 + u3);
177 
178  double finalV = denom1 * denom2;
179  finalV = numerator / finalV;
180 
181  double angle = Math.sinh(finalV);
182 
183  return angle;
184 
185  }
